41 void Adafruit_MS_PWMServoDriver::setPWMFreq(float freq) {
42  //Serial.print("Attempting to set freq ");
43  //Serial.println(freq);
44 
45  freq *= 0.9; // Correct for overshoot in the frequency setting (see issue #11).
46 
47  float prescaleval = 25000000;
48  prescaleval /= 4096;
49  prescaleval /= freq;
50  prescaleval -= 1;
51  //Serial.print("Estimated pre-scale: "); Serial.println(prescaleval);
52  uint8_t prescale = floor(prescaleval + 0.5);
53  //Serial.print("Final pre-scale: "); Serial.println(prescale);
54 
55  uint8_t oldmode = read8(PCA9685_MODE1);
56  uint8_t newmode = (oldmode&0x7F) | 0x10; // sleep
57  write8(PCA9685_MODE1, newmode); // go to sleep
58  write8(PCA9685_PRESCALE, prescale); // set the prescaler
59  write8(PCA9685_MODE1, oldmode);
60  delay(5);
61  write8(PCA9685_MODE1, oldmode | 0xa1); // This sets the MODE1 register to turn on auto increment.
62  // This is why the beginTransmission below was not working.
63  // Serial.print("Mode now 0x"); Serial.println(read8(PCA9685_MODE1), HEX);
64 }
